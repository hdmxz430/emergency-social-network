const {directory} = require('../models/esn_directory');

let loginOrRegister = (req, res) => {
    let user = _.pick(req.query, ['username', 'password']);
    console.log(user);
    User.check(user.username, user.password)
        .then(user => user.generateToken())
        .then(user => {
            let userRes = _.pick(user, ['username', 'user_status', 'status_timestamp']);
            directory.updateStatus(userRes.username, true);
            res.send({token: user.token, user: userRes});
        })
        .catch(err => defaultErr(err, res));
};
 
let register = (req, res) => {
    let user = _.pick(req.body, ['username', 'password']);
    console.log(user);
    User.Create(user)
        .then(user => user.generateToken())
        .then(user => {
            directory.addUser(user, true);
            let userRes = _.pick(user, ['username', 'user_status', 'status_timestamp']);
            res.send({token: user.token, user: userRes});
        })
        .catch(err => defaultErr(err, res));
};
 
let logout = (req, res) => {
    req.user.removeToken().then(() => {
        directory.updateStatus(req.user.username, false);
        let userList = directory.sort();
        const io = require('../app').io;
        io.emit('update', {users: userList});
        res.send({status: 200});
    }).catch(err => defaultErr(err, res));
};
